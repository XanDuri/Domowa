\section{Pakiet \texttt{listings} dla wpisywania kodów}
\lstset{language=Pascal}
\begin{lstlisting}
function power(x: integer,n: integer): integer;
Var k,a,b:integer;
Begin
	k:=n; a:=1; b:=x;
	while k>0 do begin {Niezmiennik: x^n=a*b^k}
		if k mod 2=0 then begin
			k:=k/2;
			b:=b*b;
		end else begin
			k:=k-1;
			a:=a*b;
		end;
	end;
	power:=a;
End;
\end{lstlisting}
